Password Criteria Checker



  
  
import string
from pyscript import Element

# Check for repeating characters
def has_three_repeating_characters(password):
    for i in range(len(password) - 2):
        if password[i] == password[i+1] == password[i+2]:
            return True
    return False

# Main password check logic
def check_password(password):
    unmet_criteria = []

    if len(password) < 12:
        unmet_criteria.append("Password is too short. It needs 12 or more characters.")
    if not any(c.isupper() for c in password):
        unmet_criteria.append("Password must contain an uppercase letter.")
    if not any(c.islower() for c in password):
        unmet_criteria.append("Password must contain a lowercase letter.")
    if not any(c.isdigit() for c in password):
        unmet_criteria.append("Password must contain a number.")
    if not any(c in string.punctuation for c in password):
        unmet_criteria.append("Password must contain a special character.")
    if "password" in password.lower():
        unmet_criteria.append("Password must not contain the word 'password'.")
    if has_three_repeating_characters(password):
        unmet_criteria.append("Password cannot contain repeating characters of three or more.")
    
    common_passwords = {"123456", "qwerty", "welcome", "admin", "tom's website"}
    if password.lower() in common_passwords:
        unmet_criteria.append("Password is too easy to guess.")

    if not unmet_criteria:
        return "✅ This password meets the requirements!"
    else:
        return "❌ The following criteria were not met:\n- " + "\n- ".join(unmet_criteria)

# Function called when button is clicked
def on_click(*args):
    pwd = Element("password_input").element.value
    result = check_password(pwd)
    Element("output").element.innerText = result
